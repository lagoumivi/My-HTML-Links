# 1. Normal distribution example
#Normal: Fits alcohol % → calculates probability ≤ 10 → plots PDF.

# Feature: 'alcohol'
alcohol = df['alcohol']
mu, std = norm.fit(alcohol)
print("Alcohol mean & std:", mu, std)

# Probability that alcohol ≤ 10%
prob_norm = norm.cdf(10, mu, std)
print("P(alcohol ≤ 10%) =", prob_norm)

# Plot histogram + PDF
plt.hist(alcohol, bins=20, density=True, alpha=0.6, color='g')
xmin, xmax = alcohol.min(), alcohol.max()
x = np.linspace(xmin, xmax, 100)
plt.plot(x, norm.pdf(x, mu, std), 'r', lw=2)
plt.title("Normal fit: Alcohol")
plt.show()

# 4. Exponential example
#Exponential: Uses residual sugar as “waiting time” → probability sugar > 5.
# Time between events: simulate using 'residual sugar' as proxy
sugar = df['residual sugar'] + 0.01  # avoid zeros
scale = sugar.mean()

print("Exponential P(residual sugar > 5):", prob_expon)

Rule of Thumb for Distributions

Normal (Gaussian) → Use when values are continuous, roughly bell-shaped around a mean. E.g. people’s heights, test scores, or in wine → alcohol % across bottles.

Binomial → When you have repeated yes/no trials with fixed probability. E.g. coin flips, whether a passenger on Titanic survived (success/failure).

Poisson → When you count rare events happening in a time/space window. E.g. number of lifeboats launched per hour, or number of distress signals sent.

Exponential → Models time between Poisson events. E.g. waiting time until the next passenger boards, or until the next iceberg sighting.

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import scipy.stats as stats
from io import StringIO
import requests

# --- Load Titanic dataset ---
url = "https://raw.githubusercontent.com/mwaskom/seaborn-data/master/titanic.csv"
res = requests.get(url)
df = pd.read_csv(StringIO(res.text))
ages = df['age'].dropna()

# --- Candidate distributions ---
dists = [stats.norm, stats.lognorm, stats.expon, stats.gamma]
x = np.linspace(ages.min(), ages.max(), 200)

plt.figure(figsize=(8,5))
plt.hist(ages, bins=30, density=True, alpha=0.5, color='grey', label="Age data")

# Fit each distribution and plot PDF
for dist in dists:
    params = dist.fit(ages)
    # Compute KS test
    ks_stat, pval = stats.kstest(ages, dist.cdf, args=params)
    
    # Generate PDF / PMF depending on distribution type
    if dist in [stats.norm, stats.expon, stats.gamma]:
        pdf = dist.pdf(x, *params)  # continuous
        plt.plot(x, pdf, lw=2, label=f"{dist.name} (p={pval:.3f})")
    else:  # discrete distributions like lognorm sometimes
        pdf = dist.pdf(x, *params)
        plt.plot(x, pdf, lw=2, label=f"{dist.name} (p={pval:.3f})")

plt.title("Titanic Age: Distribution Fit Comparison")
plt.xlabel("Age")
plt.ylabel("Density")
plt.legend()
plt.show()

import pandas as pd
import numpy as np
import scipy.stats as stats
from io import StringIO
import requests

# --- Load Titanic dataset ---
url = "https://raw.githubusercontent.com/mwaskom/seaborn-data/master/titanic.csv"
res = requests.get(url)
df = pd.read_csv(StringIO(res.text))

# --- Fit Gamma to existing ages ---
ages = df['age'].dropna()
gamma_params = stats.gamma.fit(ages)  # returns (shape, loc, scale)

# --- Impute missing ages by sampling from fitted Gamma ---
n_missing = df['age'].isna().sum()
imputed_ages = stats.gamma.rvs(*gamma_params, size=n_missing, random_state=42)

# Fill missing values
df.loc[df['age'].isna(), 'age'] = imputed_ages

# Check result
print("Number of missing ages now:", df['age'].isna().sum())
print(df['age'].head(15))


print("Original mean (excluding missing):", ages.mean())
print("New mean (after imputation):", df['age'].mean())
